Enter some C source code and get it tokenized!

Now supports math operations, if statements, loops, etc. Doesn't support pointers or arrays.

Here's a long demo that shows off a lot of functionality: 
static int bar = 99;
char baz;
long foobar = 2278,
     barbaz = 3389;

string myStr = "hello, this is a string.";

/* here is a C comment 
 * and here is a multiline one, is there support?
 */

unsigned long long
foobar (char arg1, int arg2)
{
    int number = 19; /* inline comment support? */
    char foo = 'b';
    if (bar != number)
    {
        number = 8;
    } else if (bar > 0) {
        number = 10;
    } else {
        number = 12;
    }
    bar *= 9;
    return number;
}

unsigned int main (int argc, char * argv[]) {
    fprintf(stdout, "Hello World\n");

    unsigned int foobaz = 0;

    switch (argc) {
        case 1:
            foobaz = number + 10;
            break;
        case 2:
            foobaz = bar + 10;
            break;
        default:
            foobaz = 10;
    }

    return foobaz;
}
